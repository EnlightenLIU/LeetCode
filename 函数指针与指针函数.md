#include <stdio.h>
//指针函数
void main()
{
    float score[][4] = {{60, 70, 80, 90}, {56, 89, 67, 88}, {34, 78, 90, 86}};
    float *search(float(*pointer)[4], int n);
    float *p;
    int i, m;
    printf("enter the number of student:");
    scanf("%d", &m);
    printf("the score of the No.%d are:", m);
    p = search(score, m);
    for (i = 0; i < 4; i++)
    {
        printf("%5.2f\t", *(p + i));
    }
    printf("\n");
}

float *search(float (*pointer)[4], int n)
{
    float *pt;
    pt = *(pointer + n);
    return (pt); //返回的是*pt=60时的地址
}

#include <stdio.h>
//函数指针
void main()
{
    int max(int, int);
    int (*p)(int, int);
    int a, b, c;
    p = max; //函数的入口地址赋给指针
    scanf("%d,%d", &a, &b);
    c = (*p)(a, b);
    printf("a=%d,b=%d,c=%d", a, b, c);
}

int max(int x, int y)
{
    int z;
    if (x > y)
        z = x;
    else
        z = y;
    return z;
}
